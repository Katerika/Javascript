let basket = [
  {
    name: 'pen',
    price: 70,
    count: 5
  },
  {
    name: 'pencil',
    price: 30,
    count: 7
  },
  {
    name: 'paper',
    price: 500,
    count: 3
  },
]

function countBasketPrice(basket){
  let basketPrice = 0;
  for (let i of basket){
    basketPrice += i.price * i.count;
  }
  return basketPrice;
}

function countBasket(basket){
  let basketCount = 0;
  for (let i of basket){
    basketCount += i.count;
  }
  return basketCount;
}

console.log(`В корзине: ${countBasket(basket)} товаров на сумму ${countBasketPrice(basket)} рублей`);
